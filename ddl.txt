CREATE TABLE REUNIAO (ID  SERIAL NOT NULL, ASSUNTO VARCHAR(255), DATAHORA TIMESTAMP, PRIMARY KEY (ID))
CREATE TABLE CONVIDADO (ID INTEGER NOT NULL, EMAIL VARCHAR(255), EMPRESA VARCHAR(255), NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE PARTICIPANTE (ID INTEGER NOT NULL, EMAIL VARCHAR(255), NOME VARCHAR(255), PRIMARY KEY (ID))
CREATE TABLE PARTICIPANTE_REUNIAO (reunioes_ID INTEGER NOT NULL, participantes_ID INTEGER NOT NULL, PRIMARY KEY (reunioes_ID, participantes_ID))
CREATE TABLE CONVIDADO_REUNIAO (Convidado_ID INTEGER NOT NULL, reunioes_ID INTEGER NOT NULL, PRIMARY KEY (Convidado_ID, reunioes_ID))
